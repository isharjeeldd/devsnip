useDebouncedValue — React hook

Lightweight hook for debouncing rapidly-changing values. Useful for search inputs, resize observers, and any derived state you don't want to recompute on every keystroke.

Edited 12m ago· Created Apr 3, 2026· TypeScript
TypeScript use-debounced-value.ts 
import { useEffect, useState } from 'react';

// Returns the latest value after `delay` ms of silence.
export function useDebouncedValue<T>(value: T, delay = 250): T {
  const [debounced, setDebounced] = useState(value);

  useEffect(() => {
    const id = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(id);
  }, [value, delay]);

  return debounced;
}
